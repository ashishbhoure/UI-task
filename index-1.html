public class SecondLargestFinder { public static int secondLargest(int[] nums) { if (nums == null || nums.length < 2) { throw new IllegalArgumentException("Array must have at least two elements"); } int first = Integer.MIN_VALUE; int second = Integer.MIN_VALUE; for (int num : nums) { if (num > first) { second = first; first = num; } else if (num > second && num < first) { second = num; } } if (second == Integer.MIN_VALUE) { throw new IllegalArgumentException("There is no second largest element"); } return second; } public static void main(String[] args) { int[] nums = {10, 5, 20, 20, 4, 5}; try { System.out.println("The second largest number is: " + secondLargest(nums)); } catch (IllegalArgumentException e) { System.out.println(e.getMessage()); } } }
